-- Write to file quickly
vim.api.nvim_set_keymap('n', 'w', ':w!', { noremap = true, silent = true })

-- Edit Neovim configuration file
vim.api.nvim_set_keymap('n', 've', ':e $MYVIMRC', { noremap = true, silent = true })

-- Source (reload) Neovim configuration file
vim.api.nvim_set_keymap('n', 'vs', ':source $MYVIMRC', { noremap = true, silent = true })
