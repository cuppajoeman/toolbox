-- Open files relative to the directory the current file lives in
vim.api.nvim_set_keymap('n', 'ce', ':e =expand("%:p:h") . "/"', { noremap = true, silent = true })
vim.api.nvim_set_keymap('n', 'ct', ':tabe =expand("%:p:h") . "/"', { noremap = true, silent = true })
vim.api.nvim_set_keymap('n', 'cs', ':split =expand("%:p:h") . "/"', { noremap = true, silent = true })
vim.api.nvim_set_keymap('n', 'cv', ':vsplit =expand("%:p:h") . "/"', { noremap = true, silent = true })
